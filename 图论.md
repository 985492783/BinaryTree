# Floyd算法
上代码:洛谷P6175：无向图的最小环
```
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    static final int Max=(int)1e8;
    public static void main(String[] args){
        Scanner in=new Scanner(System.in);
        int ans= Max;
        int n=in.nextInt();
        int m=in.nextInt();
        int[][] a=new int[n+1][n+1];
        int[][] f=new int[n+1][n+1];
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
                if(i!=j) {
                    a[i][j]=Max;
                    f[i][j]=Max;
                }
        for(int i=0;i<m;i++){
            int u=in.nextInt();
            int v=in.nextInt();
            int d=in.nextInt();
            a[u][v]=d;a[v][u]=d;
            f[u][v]=d;f[v][u]=d;
        }
        for(int k=1;k<=n;k++){
            for(int i=1;i<k;i++)
                for(int j=i+1;j<k;j++)
                    ans=Math.min(ans,f[i][j]+a[i][k]+a[k][j]);
            for(int i=1;i<=n;i++)
                for(int j=1;j<=n;j++){
                    f[i][j]=Math.min(f[i][j],f[i][k]+f[k][j]);
                    f[j][i]=f[i][j];
                }
        }
        if(ans==Max)System.out.println("No solution.");
        else System.out.println(ans);
    }
}
```
