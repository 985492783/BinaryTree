# Floyd算法
上代码:洛谷P6175：无向图的最小环
```
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    static final int Max=(int)1e8;
    public static void main(String[] args){
        Scanner in=new Scanner(System.in);
        int ans= Max;
        int n=in.nextInt();
        int m=in.nextInt();
        int[][] a=new int[n+1][n+1];
        int[][] f=new int[n+1][n+1];
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
                if(i!=j) {
                    a[i][j]=Max;
                    f[i][j]=Max;
                }
        for(int i=0;i<m;i++){
            int u=in.nextInt();
            int v=in.nextInt();
            int d=in.nextInt();
            a[u][v]=d;a[v][u]=d;
            f[u][v]=d;f[v][u]=d;
        }
        for(int k=1;k<=n;k++){
            for(int i=1;i<k;i++)
                for(int j=i+1;j<k;j++)
                    ans=Math.min(ans,f[i][j]+a[i][k]+a[k][j]);
            for(int i=1;i<=n;i++)
                for(int j=1;j<=n;j++){
                    f[i][j]=Math.min(f[i][j],f[i][k]+f[k][j]);
                    f[j][i]=f[i][j];
                }
        }
        if(ans==Max)System.out.println("No solution.");
        else System.out.println(ans);
    }
}
```

POJ-2387——无向图的最短路
```
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    static final int Max= (int) 1e8;
    public static void main(String[] args){
        Scanner in=new Scanner(System.in);
        int T=in.nextInt();
        int N=in.nextInt();
        int[][] map=new int[N+1][N+1];
        for(int i=1;i<=N;i++)
            for(int j=1;j<=N;j++)
                if(i!=j)
                    map[i][j]=Max;
        int[] dis=new int[N+1];
        Arrays.fill(dis,Max);dis[N]=0;
        boolean[] p=new boolean[N+1];p[0]=true;
        while(T-->0){
            int u=in.nextInt();
            int v=in.nextInt();
            int d=in.nextInt();
            map[u][v]=Math.min(map[u][v],d);
            map[v][u]=Math.min(map[v][u],d);
        }
        int i=N;
        while(!p[i]){
            p[i]=true;
            int min=Integer.MAX_VALUE;
            int minIndex=0;
            for(int j=1;j<=N;j++){
                if(!p[j]){
                    dis[j]=Math.min(dis[j],dis[i]+map[i][j]);
                    if(min>dis[j]){
                        min=dis[j];
                        minIndex=j;
                    }
                }
            }
            i=minIndex;
        }
        System.out.println(dis[1]);
    }
}

POJ-3268——有向图的最短路  
1-x与x-1的有向图最短路实质将x-1的路径方向反转(在读入时即可实现)
```
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    static final int Max= (int) 1e8;
    public static void main(String[] args){
        Scanner in=new Scanner(System.in);
        int N=in.nextInt();
        int M=in.nextInt();
        int X=in.nextInt();
        int[][] arr=new int[N+1][N+1];
        int[][] brr=new int[N+1][N+1];
        for(int i=1;i<=N;i++)
            for(int j=1;j<=N;j++)
                if(i!=j) arr[i][j]=brr[i][j]=Max;
        int[] dis=new int[N+1];
        int[] dis2=new int[N+1];
        boolean[] p=new boolean[N+1];
        boolean[] q=new boolean[N+1];
        Arrays.fill(dis,Max);dis[X]=0;
        Arrays.fill(dis2,Max);dis2[X]=0;
        while(M-->0){
            int u=in.nextInt();
            int v=in.nextInt();
            int d=in.nextInt();
            arr[u][v]=Math.min(d,arr[u][v]);
            brr[v][u]=Math.min(d,brr[v][u]);
        }
        int i=X;
        while(!p[i]){
            p[i]=true;
            int min=Integer.MAX_VALUE;
            int minIndex=0;
            for(int j=1;j<=N;j++)
                if(!p[j]){
                    dis[j]=Math.min(dis[i]+arr[i][j],dis[j]);
                    if(min>dis[j]){
                        min=dis[j];
                        minIndex=j;
                    }
                }
            i=minIndex;
        }
        i=X;
        while(!q[i]){
            q[i]=true;
            int min=Integer.MAX_VALUE;
            int minIndex=0;
            for(int j=1;j<=N;j++)
                if(!q[j]){
                    dis2[j]=Math.min(dis2[i]+brr[i][j],dis2[j]);
                    if(min>dis2[j]){
                        min=dis2[j];
                        minIndex=j;
                    }
                }
            i=minIndex;
        }
        int max=0;
        for(int j=1;j<=N;j++)
            if(j!=X)
            max=Math.max(max,dis[j]+dis2[j]);
        System.out.println(max);
    }
}
